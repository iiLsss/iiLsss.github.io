// Curried with placeholders.
curried(1)(_, 3)(2);
// => [1, 2, 3]
function curry(func) {
  let curried = (...args) => {
    if (func.length > args.length) {
      return (...reset) => curried(...args, ...reset);
    }
    return func(...args);
  };
  return curried;
}
